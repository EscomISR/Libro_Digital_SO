InfoSistema.bat 
@echo off
:: Escuela: ESCOM - IPN
:: Materia: Sistemas Operativos
:: Unidad 1: Introducción a los Sistemas Operativos
:: Descripción: Script básico de información del sistema en Batch

title Mi Primer Script de SO
color 0A

echo ==============================================
echo       BIENVENIDO AL SISTEMA OPERATIVO
echo ==============================================
echo.

:: Muestra el nombre del usuario actual
echo Hola, usuario actual: %USERNAME%
echo.

:: Muestra la fecha y hora del sistema
echo Fecha del sistema:
date /t
echo Hora del sistema:
time /t
echo.

:: Información de la versión de Windows
echo Versión del Sistema Operativo:
ver
echo.

echo ==============================================
echo Listando archivos en el directorio actual:
echo ==============================================
:: Comando para listar archivos (formato simple)
dir /b

echo.
echo Fin del script. Presione una tecla para salir.
pause > nul
