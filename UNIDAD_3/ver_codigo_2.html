MemoriaSegmentada.c 
/*
    Escuela: ESCOM - IPN
    Materia: Sistemas Operativos
    Práctica 9: Memoria Segmentada
    Descripción: Simulación de traducción de direcciones lógicas a físicas
                 usando tabla de segmentos (Base + Límite).
*/
#include <stdio.h>
#include <stdlib.h>

#define NUM_SEGMENTOS 5

// Estructura que define un segmento en la tabla
typedef struct {
    int base;   // Dirección base física
    int limite; // Tamaño del segmento (Offset máximo permitido)
} Segmento;

// Tabla de segmentos simulada
Segmento tabla_segmentos[NUM_SEGMENTOS] = {
    {1000, 400},  // Segmento 0: Código
    {2400, 500},  // Segmento 1: Datos Globales
    {1800, 200},  // Segmento 2: Pila (Stack)
    {3500, 100},  // Segmento 3: Heap
    {4200, 300}   // Segmento 4: Extra
};

// Función para traducir dirección lógica a física
void acceder_memoria(int num_segmento, int offset) {
    printf("Solicitud -> Segmento: %d | Desplazamiento: %d\n", num_segmento, offset);

    // 1. Validar que el segmento exista
    if (num_segmento < 0 || num_segmento >= NUM_SEGMENTOS) {
        printf("  [ERROR] Segmento inválido (Trap de direccionamiento)\n\n");
        return;
    }

    // 2. Validar el desplazamiento contra el límite
    if (offset >= tabla_segmentos[num_segmento].limite) {
        printf("  [ERROR] Fallo de Segmentación (Offset > Límite)\n");
        printf("  Límite del segmento %d es %d\n\n", num_segmento, tabla_segmentos[num_segmento].limite);
        return;
    }

    // 3. Calcular dirección física
    int direccion_fisica = tabla_segmentos[num_segmento].base + offset;
    printf("  [OK] Dirección Física = Base (%d) + Offset (%d) = %d\n\n", 
           tabla_segmentos[num_segmento].base, offset, direccion_fisica);
}

int main() {
    printf("--- Simulación de Memoria Segmentada ---\n\n");

    // Pruebas de acceso
    acceder_memoria(0, 100);  // Válido
    acceder_memoria(1, 600);  // Inválido (Excede límite 500)
    acceder_memoria(2, 50);   // Válido
    acceder_memoria(5, 10);   // Segmento inexistente
    acceder_memoria(3, 99);   // Válido (Límite)

    return 0;
}
