/*
    Escuela: ESCOM - IPN
    Materia: Sistemas Operativos
    Práctica 1: Memoria Virtual
    Descripción: Simulación básica de paginación y fallos de página (FIFO)
*/
#include <stdio.h>
#include <stdlib.h>

#define MAX_MARCOS 3      // Número de marcos de página en memoria física
#define MAX_PAGINAS 20    // Longitud de la cadena de referencia

void mostrar_marcos(int marcos[], int n) {
    for (int i = 0; i < n; i++) {
        if (marcos[i] == -1)
            printf("[ ] ");
        else
            printf("[%d] ", marcos[i]);
    }
    printf("\n");
}

int main() {
    // Cadena de referencia de páginas solicitadas por el proceso
    int paginas[MAX_PAGINAS] = {7, 0, 1, 2, 0, 3, 0, 4, 2, 3, 0, 3, 2, 1, 2, 0, 1, 7, 0, 1};
    int marcos[MAX_MARCOS];
    int fallos_pagina = 0;
    int puntero = 0; // Para algoritmo FIFO
    
    // Inicializar marcos vacíos
    for (int i = 0; i < MAX_MARCOS; i++) {
        marcos[i] = -1;
    }

    printf("--- Simulación de Reemplazo de Página (FIFO) ---\n");
    printf("Cadena de referencia: ");
    for(int i=0; i<MAX_PAGINAS; i++) printf("%d ", paginas[i]);
    printf("\n\n");

    for (int i = 0; i < MAX_PAGINAS; i++) {
        int pagina_actual = paginas[i];
        int encontrada = 0;

        // Verificar si la página ya está en memoria (HIT)
        for (int j = 0; j < MAX_MARCOS; j++) {
            if (marcos[j] == pagina_actual) {
                encontrada = 1;
                break;
            }
        }

        printf("Ref %d: ", pagina_actual);

        if (encontrada) {
            printf("HIT  -> ");
        } else {
            // Fallo de página (MISS) -> Reemplazar
            marcos[puntero] = pagina_actual;
            puntero = (puntero + 1) % MAX_MARCOS; // Avanzar puntero FIFO circular
            fallos_pagina++;
            printf("MISS -> ");
        }
        
        mostrar_marcos(marcos, MAX_MARCOS);
    }

    printf("\nTotal de fallos de página: %d\n", fallos_pagina);
    printf("Rendimiento (Hits): %d/%d\n", (MAX_PAGINAS - fallos_pagina), MAX_PAGINAS);

    return 0;
}
