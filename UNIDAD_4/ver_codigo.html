#include <stdio.h>
#include <dirent.h>
#include <string.h>
#include <limits.h>
#include <sys/stat.h>

void mostrar_directorios(const char *ruta, int nivel) {
    DIR *dir;                      // Puntero al directorio
    struct dirent *entrada;        // Estructura para almacenar información de la entrada
    struct stat info;              // Estructura para almacenar información del archivo

    if ((dir = opendir(ruta)) == NULL) {   // Abrir el directorio especificado en 'ruta'
        perror("Error al abrir el directorio");   // Imprimir un mensaje de error si no se puede abrir
        return;
    }

    while ((entrada = readdir(dir)) != NULL) {   // Leer cada entrada del directorio
        char ruta_completa[PATH_MAX];   // Almacenar la ruta completa del archivo/directorio

        snprintf(ruta_completa, PATH_MAX, "%s/%s", ruta, entrada->d_name);   // Construir la ruta completa

        if (strcmp(entrada->d_name, ".") == 0 || strcmp(entrada->d_name, "..") == 0) {
            continue;   // Ignorar las entradas '.' y '..' (directorio actual y directorio padre)
        }

        if (stat(ruta_completa, &info) == -1) {   // Obtener información del archivo/directorio
            perror("Error al obtener información del archivo");   // Imprimir un mensaje de error si no se puede obtener
            continue;
        }

        printf("%*s", nivel * 4, "");   // Imprimir espacios en blanco para la indentación
        printf("%s\n", entrada->d_name);   // Imprimir el nombre de la entrada

        if (S_ISDIR(info.st_mode)) {   // Verificar si es un directorio
            mostrar_directorios(ruta_completa, nivel + 1);   // Llamada recursiva para mostrar los subdirectorios
        }
    }

    closedir(dir);   // Cerrar el directorio
}

int main() {
    // Nota: Se recomienda cambiar esta ruta por una válida en tu sistema local
    const char *ruta_base = ".";   
    mostrar_directorios(ruta_base, 0);   // Llamar a la función para mostrar los directorios y subdirectorios

    return 0;
}
