Software de E/S y Señales
SoftwareIO.c 
/*
    Escuela: ESCOM - IPN
    Materia: Sistemas Operativos
    Unidad 5: Entrada y Salida
    Archivo: SoftwareIO.c
    Descripción: Principios del Software de E/S.
                 Demostración de E/S Programada, Interrupciones (Señales) y DMA.
*/
#include <stdio.h> 
#include <stdlib.h> 
#include <string.h> 
#include <unistd.h>  // Para simular retardos y salidas 
#include <signal.h>  // Para manejo de interrupciones 
#include <time.h> 

// Prototipos de funciones 
void esProgramada(); 
void esInterrupciones(); 
void esDMA(); 
void handlerSIGINT(int sig); 

// Variable global para simular buffer DMA 
char dmaBuffer[1024]; 

int main() { 
    printf("=== Principios del Software de E/S ===\n"); 
    
    // 1. E/S Programadas 
    printf("\n1. E/S Programadas:\n"); 
    esProgramada(); 
    
    // 2. E/S Manejadas por Interrupciones 
    printf("\n2. E/S Manejadas por Interrupciones:\n"); 
    signal(SIGINT, handlerSIGINT); // Configura interrupción 
    esInterrupciones(); 
    
    // 3. E/S usando DMA 
    printf("\n3. E/S usando DMA:\n"); 
    esDMA(); 
    
    printf("\n=== Fin del Programa ===\n"); 
    return 0; 
} 

// 1. E/S Programadas 
void esProgramada() { 
    char data[100]; 
    printf("Escribe algo (E/S programada): "); 
    // La CPU espera aquí activamente hasta que el usuario ingresa datos
    fgets(data, sizeof(data), stdin); 
    printf("Leído: %s", data); 
    printf("La CPU espera activamente hasta que la operación se completa.\n"); 
} 

// 2. E/S Manejadas por Interrupciones 
void esInterrupciones() { 
    printf("Presiona Ctrl+C para simular una interrupción.\n"); 
    printf("La CPU puede realizar otras tareas mientras espera...\n"); 
    
    // Simulación de espera activa realizando "otras tareas"
    // En un sistema real, el proceso se bloquearía o la CPU ejecutaría otro proceso
    for (int i = 0; i < 5; i++) { 
        printf("Realizando otra tarea %d...\n", i + 1); 
        sleep(1); // Simula otra tarea 
    } 
} 

// Manejador de señal SIGINT (interrupción) 
void handlerSIGINT(int sig) { 
    printf("\n¡Interrupción recibida! Código de señal: %d\n", sig); 
    printf("La CPU responde a la interrupción y realiza la operación de E/S.\n"); 
    // Restaurar el manejador por defecto o salir si se desea
    exit(0); 
} 

// 3. E/S usando DMA 
void esDMA() { 
    printf("Iniciando simulación de E/S con DMA...\n"); 
    
    // Simula transferencia DMA 
    strcpy(dmaBuffer, "Datos transferidos vía DMA."); 
    
    sleep(1); // Simula tiempo de transferencia independiente de la CPU
    
    printf("Contenido en el buffer DMA: %s\n", dmaBuffer); 
    printf("La CPU delega la transferencia de datos al controlador DMA.\n"); 
    printf("La CPU está libre para realizar otras tareas mientras tanto.\n"); 
}
