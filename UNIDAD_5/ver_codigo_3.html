E/S Bloqueante y Buffering
BlockingIO.c 
/*
    Escuela: ESCOM - IPN
    Materia: Sistemas Operativos
    Unidad 5: Entrada y Salida
    Archivo: BlockingIO.c
    Descripción: Simulación de llamadas al sistema bloqueantes para 
                 teclado y disco, y el uso de buffers.
*/
#include <stdio.h> 
#include <stdlib.h> 
#include <string.h> 

// Prototipos de funciones 
void manejador_interrupciones(char dispositivo); // Simula el manejo de IRQ
void controlador_teclado(const char* buffer);    // Procesa entrada del teclado 
void controlador_disco(int dato);                // Procesa datos del disco 

int main() { 
    char entrada[50]; // Buffer para almacenar texto del usuario 
    int opcion, dato; // Variables para la opción y datos numéricos
 
    printf("=== Sistema de Gestión de E/S ===\n");
    printf("Seleccione una operación:\n"); 
    printf("1. Ingresar texto desde el teclado (E/S Caracteres)\n"); 
    printf("2. Guardar un número en el disco (E/S Bloques)\n"); 
    printf("Opción: "); 
    
    // scanf es una operación bloqueante: el proceso se detiene hasta recibir input
    scanf("%d", &opcion);  
    getchar(); // Limpia el buffer de entrada (stdin)
 
    switch (opcion) { 
        case 1: 
            printf("Escriba un texto: "); 
            // fgets bloquea el proceso hasta que hay un salto de línea
            fgets(entrada, sizeof(entrada), stdin); 
            
            // Procesamiento del buffer
            entrada[strcspn(entrada, "\n")] = '\0'; 
            
            // Simulación del ciclo de interrupción
            manejador_interrupciones('T'); 
            controlador_teclado(entrada); 
            break; 
            
        case 2: 
            printf("Ingrese un número a guardar: "); 
            scanf("%d", &dato); 
            
            // Simulación del ciclo de interrupción
            manejador_interrupciones('D'); 
            controlador_disco(dato); 
            break; 
            
        default: 
            printf("Opción inválida.\n"); 
            break; 
    } 
 
    return 0; 
} 
 
// Función que simula un manejador de interrupciones (ISR)
void manejador_interrupciones(char dispositivo) { 
    if (dispositivo == 'T') { 
        printf("\n[IRQ 1] Interrupción: Procesando entrada del teclado...\n"); 
    } else if (dispositivo == 'D') { 
        printf("\n[IRQ 14] Interrupción: Preparando para guardar en el disco...\n"); 
    } else { 
        printf("\n[IRQ ?] Interrupción: Dispositivo desconocido.\n"); 
    } 
} 
 
// Función que simula un controlador de teclado 
void controlador_teclado(const char* buffer) { 
    printf("Controlador del teclado: Moviendo datos del buffer al espacio de usuario.\n");
    printf("Texto recibido -> \"%s\"\n", buffer); 
} 

// Función que simula un controlador de disco 
void controlador_disco(int dato) { 
    printf("Controlador del disco: Escribiendo bloque en el sector lógico.\n");
    printf("Guardando el número %d en el disco...\n", dato); 
    printf("Operación de E/S completada con éxito.\n"); 
}
