#include <stdio.h>
#include <stdlib.h>
#include <time.h>

// Estructura para representar un proceso
typedef struct {
    int id;
    int bt;
    int wt;
    int tat;
} Proceso;

// Estructura para representar una cola tipo FIFO
typedef struct Nodo {
    Proceso proceso;
    struct Nodo* siguiente;
} Nodo;

typedef struct {
    Nodo* frente;
    Nodo* final;
} Cola;

// Función para crear un nuevo nodo
Nodo* crearNodo(Proceso proceso) {
    Nodo* nodo = (Nodo*)malloc(sizeof(Nodo));
    nodo->proceso = proceso;
    nodo->siguiente = NULL;
    return nodo;
}

// Función para encolar un proceso a la cola
void encolar(Cola* cola, Proceso proceso) {
    Nodo* nodo = crearNodo(proceso);
    if (cola->frente == NULL) {
        cola->frente = nodo;
        cola->final = nodo;
    } else {
        cola->final->siguiente = nodo;
        cola->final = nodo;
    }
}

// Función para desencolar un proceso de la cola
Proceso desencolar(Cola* cola) {
    if (cola->frente == NULL) {
        Proceso proceso_vacio;
        proceso_vacio.id = -1;
        return proceso_vacio;
    } else {
        Nodo* temp = cola->frente;
        Proceso proceso = temp->proceso;
        cola->frente = cola->frente->siguiente;
        free(temp);
        return proceso;
    }
}

// Función para liberar la memoria de la cola
void liberarCola(Cola* cola) {
    Nodo* actual = cola->frente;
    while (actual != NULL) {
        Nodo* temp = actual;
        actual = actual->siguiente;
        free(temp);
    }
    cola->frente = NULL;
    cola->final = NULL;
}

// Función para calcular los tiempos de espera y de retorno promedio
void calcularTiempos(Proceso procesos[], int n) {
    int tiempo_espera_total = 0;
    int tiempo_retorno_total = 0;

    for (int i = 0; i < n; i++) {
        tiempo_espera_total += procesos[i].wt;
        tiempo_retorno_total += procesos[i].tat;
    }

    double tiempo_espera_promedio = (double)tiempo_espera_total / n;
    double tiempo_retorno_promedio = (double)tiempo_retorno_total / n;

    // Mostrar la tabla de resultados en un archivo de texto
    FILE* archivo_resultados = fopen("resultados_fcfs.txt", "w");
    if (archivo_resultados != NULL) {
        fprintf(archivo_resultados, "Proceso\tTiempo de espera\tTiempo de retorno\n");
        for (int i = 0; i < n; i++) {
            fprintf(archivo_resultados, "%d\t\t%d\t\t\t%d\n", procesos[i].id, procesos[i].wt, procesos[i].tat);
        }
        fprintf(archivo_resultados, "\nTiempo de espera promedio: %.2f\n", tiempo_espera_promedio);
        fprintf(archivo_resultados, "Tiempo de retorno promedio: %.2f\n", tiempo_retorno_promedio);
        fclose(archivo_resultados);
        printf("Se han guardado los resultados en el archivo resultados_fcfs.txt\n");
    } else {
        printf("Error al abrir el archivo de resultados.\n");
    }
}

// Función principal
int main(int argc, char* argv[]) {
    if (argc != 4) {
        printf("Uso: ./fcfs   \n");
        return 1;
    }

    int n = atoi(argv[1]); // Número de procesos
    int rango_minimo = atoi(argv[2]); // Rango mínimo de ráfaga
    int rango_maximo = atoi(argv[3]); // Rango máximo de ráfaga

    if (n <= 0) {
        printf("El número de procesos debe ser mayor a cero.\n");
        return 1;
    }

    if (rango_minimo <= 0 || rango_maximo <= 0 || rango_minimo > rango_maximo) {
        printf("El rango de ráfaga debe ser válido.\n");
        return 1;
    }

    srand(time(NULL));

    Proceso* procesos = (Proceso*)malloc(n * sizeof(Proceso)); // Crear arreglo de procesos

    // Generar procesos y asignar tiempos de ráfaga aleatorios
    for (int i = 0; i < n; i++) {
        procesos[i].id = i + 1;
        procesos[i].bt = rand() % (rango_maximo - rango_minimo + 1) + rango_minimo;
        procesos[i].wt = 0;
        procesos[i].tat = 0;
    }

    // Asignar tiempo de espera 0 a un proceso de forma aleatoria
    int indice_tiempo_espera_cero = rand() % n;
    procesos[indice_tiempo_espera_cero].wt = 0;

    // Asignar tiempos de espera aleatorios a los demás procesos
    for (int i = 0; i < n; i++) {
        if (i != indice_tiempo_espera_cero) {
            procesos[i].wt = rand() % rango_maximo + 1;
        }
    }

    Cola cola;
    cola.frente = NULL;
    cola.final = NULL;

    // Encolar los procesos en orden de llegada
    for (int i = 0; i < n; i++) {
        encolar(&cola, procesos[i]);
    }

    int tiempo_transcurrido = 0;

    // Calcular los tiempos de espera y de retorno
    while (cola.frente != NULL) {
        Proceso proceso_actual = desencolar(&cola);
        proceso_actual.tat = proceso_actual.wt + proceso_actual.bt;
        tiempo_transcurrido += proceso_actual.bt;

        // Actualizar tiempos de espera de los procesos en espera
        Nodo* actual = cola.frente;
        while (actual != NULL) {
            actual->proceso.wt += proceso_actual.bt;
            actual = actual->siguiente;
        }

        // Guardar los cambios en el proceso actual
        procesos[proceso_actual.id - 1] = proceso_actual;
    }

    liberarCola(&cola); // Liberar memoria de la cola
    calcularTiempos(procesos, n); // Calcular y mostrar los tiempos de espera y de retorno
    free(procesos); // Liberar memoria del arreglo de procesos
    return 0;
}
