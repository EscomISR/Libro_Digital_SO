#include <stdio.h>
#include <stdlib.h>

#define MAX_PROCESOS 10

struct Proceso {
    int id;
    int rafaga;          // Tiempo de ejecución necesario
    int tiempo_restante; // Tiempo que le falta por ejecutar
    int tiempo_espera;
};

int main() {
    int n, quantum;
    struct Proceso procesos[MAX_PROCESOS];
    int tiempo_total = 0, procesos_completados = 0;
    int i;

    // Configuración inicial (simulada)
    n = 4; // Número de procesos
    quantum = 2; // Tiempo del Quantum

    // Inicialización de procesos
    int rafagas[] = {5, 3, 1, 4};
    for(i = 0; i < n; i++) {
        procesos[i].id = i + 1;
        procesos[i].rafaga = rafagas[i];
        procesos[i].tiempo_restante = rafagas[i];
        procesos[i].tiempo_espera = 0;
    }

    printf("--- Simulación Round Robin (Quantum = %d) ---\n", quantum);

    // Ciclo principal de planificación
    while (procesos_completados < n) {
        int algun_proceso_ejecutado = 0;

        for (i = 0; i < n; i++) {
            if (procesos[i].tiempo_restante > 0) {
                algun_proceso_ejecutado = 1;
                printf("\nEjecutando Proceso %d... ", procesos[i].id);

                if (procesos[i].tiempo_restante > quantum) {
                    // El proceso usa todo el quantum y aún le falta
                    tiempo_total += quantum;
                    procesos[i].tiempo_restante -= quantum;
                    printf("Queda pendiente (%d restan)", procesos[i].tiempo_restante);
                } else {
                    // El proceso termina dentro de este quantum
                    tiempo_total += procesos[i].tiempo_restante;
                    procesos[i].tiempo_espera = tiempo_total - procesos[i].rafaga;
                    procesos[i].tiempo_restante = 0;
                    procesos_completados++;
                    printf("Terminado! (Espera: %d)", procesos[i].tiempo_espera);
                }
            }
        }
        
        if (!algun_proceso_ejecutado) break;
    }

    printf("\n\n--- Estadísticas Finales ---\n");
    float promedio_espera = 0;
    for (i = 0; i < n; i++) {
        promedio_espera += procesos[i].tiempo_espera;
        printf("Proceso %d: Espera = %d\n", procesos[i].id, procesos[i].tiempo_espera);
    }
    printf("Tiempo Promedio de Espera: %.2f\n", promedio_espera / n);

    return 0;
}
