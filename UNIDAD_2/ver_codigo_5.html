/*
    Escuela: ESCOM - IPN
    Materia: Sistemas Operativos
    Práctica 6: Planificación SRTF (Shortest Remaining Time First)
    Descripción: Versión expulsiva del SJF. Si llega un proceso con menor
                 tiempo restante, se apropia de la CPU.
*/

#include <stdio.h>
#include <stdbool.h>
#include <limits.h>

struct Proceso {
    int pid;
    int bt;  // Burst Time (Ráfaga)
    int art; // Arrival Time (Llegada)
};

void encontrarTiempoEspera(struct Proceso proc[], int n, int wt[]) {
    int rt[n]; // Tiempo restante (Remaining Time)
    for (int i = 0; i < n; i++)
        rt[i] = proc[i].bt;

    int complete = 0, t = 0, minm = INT_MAX;
    int shortest = 0, finish_time;
    bool check = false;

    // Ejecutar hasta que todos los procesos se completen
    while (complete != n) {
        // Encontrar proceso con tiempo restante mínimo entre los que han llegado hasta el tiempo 't'
        for (int j = 0; j < n; j++) {
            if ((proc[j].art <= t) && (rt[j] < minm) && rt[j] > 0) {
                minm = rt[j];
                shortest = j;
                check = true;
            }
        }

        if (check == false) {
            t++;
            continue;
        }

        // Reducir tiempo restante del proceso seleccionado
        rt[shortest]--;

        // Actualizar el mínimo
        minm = rt[shortest];
        if (minm == 0)
            minm = INT_MAX;

        // Si el proceso termina
        if (rt[shortest] == 0) {
            complete++;
            check = false;
            finish_time = t + 1;
            // Calcular tiempo de espera
            wt[shortest] = finish_time - proc[shortest].bt - proc[shortest].art;
            if (wt[shortest] < 0)
                wt[shortest] = 0;
        }
        t++;
    }
}

void encontrarTiempoPromedio(struct Proceso proc[], int n) {
    int wt[n], tat[n], total_wt = 0, total_tat = 0;

    encontrarTiempoEspera(proc, n, wt);

    // Calcular Turn Around Time (TAT)
    for (int i = 0; i < n; i++)
        tat[i] = proc[i].bt + wt[i];

    printf("PID\tRáfaga\tLlegada\tEspera\tRetorno\n");
    for (int i = 0; i < n; i++) {
        total_wt = total_wt + wt[i];
        total_tat = total_tat + tat[i];
        printf("%d\t%d\t%d\t%d\t%d\n", proc[i].pid, proc[i].bt, proc[i].art, wt[i], tat[i]);
    }

    printf("\nPromedio de espera = %.2f", (float)total_wt / (float)n);
    printf("\nPromedio de retorno = %.2f", (float)total_tat / (float)n);
}

int main() {
    struct Proceso proc[] = {{1, 6, 1}, {2, 8, 1}, {3, 7, 2}, {4, 3, 3}};
    int n = sizeof(proc) / sizeof(proc[0]);

    printf("--- Simulación SRTF ---\n\n");
    encontrarTiempoPromedio(proc, n);
    return 0;
}
