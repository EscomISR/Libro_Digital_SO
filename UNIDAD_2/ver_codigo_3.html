Planificación por Prioridad
PlanificarPrioridad.c 
/*
    Escuela: ESCOM - IPN
    Materia: Sistemas Operativos
    Práctica 3: Planificación por Prioridades (No Expulsivo)
    Descripción: Simula la ejecución de procesos ordenados por prioridad.
                 (Menor número = Mayor prioridad)
*/

#include <stdio.h>
#include <stdlib.h>

// Estructura que representa un proceso
typedef struct {
    int id;         // Identificador del proceso
    int rafaga;     // Tiempo de ráfaga (Burst Time)
    int prioridad;  // Nivel de prioridad
    int espera;     // Tiempo de espera
    int retorno;    // Tiempo de retorno (Turnaround Time)
} Proceso;

// Función para intercambiar dos procesos (usada en ordenamiento)
void swap(Proceso *a, Proceso *b) {
    Proceso temp = *a;
    *a = *b;
    *b = temp;
}

// Función para ordenar procesos por prioridad (burbuja)
// Nota: En este ejemplo, menor valor de 'prioridad' = mayor preferencia
void ordenarPorPrioridad(Proceso p[], int n) {
    for (int i = 0; i < n - 1; i++) {
        for (int j = 0; j < n - i - 1; j++) {
            if (p[j].prioridad > p[j + 1].prioridad) {
                swap(&p[j], &p[j + 1]);
            }
        }
    }
}

int main() {
    int n = 5; // Número de procesos
    Proceso p[5] = {
        {1, 10, 3, 0, 0},
        {2, 1, 1, 0, 0},
        {3, 2, 4, 0, 0},
        {4, 1, 5, 0, 0},
        {5, 5, 2, 0, 0}
    };

    printf("--- Planificación por Prioridades ---\n");
    printf("(Menor valor = Mayor prioridad)\n\n");

    // Ordenar la cola de listos
    ordenarPorPrioridad(p, n);

    // Calcular tiempos
    int tiempo_total = 0;
    float promedio_espera = 0;
    float promedio_retorno = 0;

    printf("Orden de Ejecución:\n");
    printf("ID\tPrioridad\tRáfaga\tEspera\tRetorno\n");

    for (int i = 0; i < n; i++) {
        p[i].espera = tiempo_total; // El tiempo que ha esperado es el total acumulado
        p[i].retorno = p[i].espera + p[i].rafaga;
        tiempo_total += p[i].rafaga;

        promedio_espera += p[i].espera;
        promedio_retorno += p[i].retorno;

        printf("P%d\t%d\t\t%d\t%d\t%d\n", 
               p[i].id, p[i].prioridad, p[i].rafaga, p[i].espera, p[i].retorno);
    }

    printf("\nPromedio de Espera: %.2f\n", promedio_espera / n);
    printf("Promedio de Retorno: %.2f\n", promedio_retorno / n);

    return 0;
}
