Sincronización (Mutex)
Mutex.c 
/*
    Escuela: ESCOM - IPN
    Materia: Sistemas Operativos
    Práctica 8: Sincronización de Procesos (Mutex)
    Descripción: Uso de Mutex para proteger una variable compartida (Sección Crítica)
                 accedida concurrentemente por múltiples hilos.
*/

#include <stdio.h>
#include <stdlib.h>
#include <pthread.h>
#include <unistd.h>

#define NUM_HILOS 5

// Recurso compartido
int saldo_bancario = 1000;

// Declaración del Mutex
pthread_mutex_t candado;

void* realizar_transaccion(void* arg) {
    int id = *((int*)arg);
    int monto = 100;

    printf("Hilo %d intentando entrar a la sección crítica...\n", id);

    // 1. Bloquear el mutex antes de entrar a la sección crítica
    pthread_mutex_lock(&candado);

    // --- INICIO SECCIÓN CRÍTICA ---
    printf("Hilo %d ha entrado. Saldo actual: %d\n", id, saldo_bancario);
    
    // Simular tiempo de procesamiento
    sleep(1);
    
    saldo_bancario += monto;
    printf("Hilo %d ha depositado $%d. Nuevo saldo: %d\n", id, monto, saldo_bancario);
    // --- FIN SECCIÓN CRÍTICA ---

    // 2. Desbloquear el mutex al salir
    pthread_mutex_unlock(&candado);

    printf("Hilo %d ha salido de la sección crítica.\n", id);
    return NULL;
}

int main() {
    pthread_t hilos[NUM_HILOS];
    int ids[NUM_HILOS];

    // Inicializar el mutex
    if (pthread_mutex_init(&candado, NULL) != 0) {
        printf("Fallo al inicializar mutex\n");
        return 1;
    }

    printf("--- Simulación de Cajero (Saldo Inicial: %d) ---\n", saldo_bancario);

    // Crear hilos
    for (int i = 0; i < NUM_HILOS; i++) {
        ids[i] = i + 1;
        pthread_create(&hilos[i], NULL, realizar_transaccion, &ids[i]);
    }

    // Esperar a que terminen los hilos
    for (int i = 0; i < NUM_HILOS; i++) {
        pthread_join(hilos[i], NULL);
    }

    // Destruir el mutex
    pthread_mutex_destroy(&candado);

    printf("\nSaldo Final: %d\n", saldo_bancario);
    return 0;
}
